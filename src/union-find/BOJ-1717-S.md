import sys
input = sys.stdin.readline

def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])
    return parent[x]

def union(a, b):
    ra = find(a)
    rb = find(b)

    if ra == rb:
        return 
    
    if rank[ra] < rank[rb]:
        parent[ra] = rb
    else:
        parent[rb] = ra
        if rank[ra] == rank[rb]:
            rank[ra] += 1 

def same_set(a, b):
    return find(a) == find(b)

N, M = map(int, input().split())
parent = list(range(N+1))
rank = [0] * (N+1)

ans = []
for _ in range(M):
    op, a, b = map(int, input().split())
    if op == 0:
        union(a,b)
    else:
        ans.append("YES" if same_set(a,b) else "NO")

sys.stdout.write("\n".join(ans))
