```python
import heapq

N = int(input())
class_lists = []

for _ in range(N):
    class_lists.append(list(map(int, input().split())))

class_lists.sort(key=lambda x:x[0])

heap = []

heapq.heappush(heap, class_lists[0][1])

for i in range(1, N):
    start, end = class_lists[i]


    if heap[0] <= start:
        heapq.heappop(heap)

    heapq.heappush(heap, end)

print(len(heap))
```
