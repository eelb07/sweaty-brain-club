N = int(input())
S = []
B = []

for _ in range(N):
    s, b = map(int, input().split())
    S.append(s)
    B.append(b)

min_val = float('inf')

def dfs(idx, curr_s, curr_b, used):
    global min_val
    if idx == N:
        if used:  # μ μ–΄λ„ ν•λ‚μ μ¬λ£λ¥Ό μ‚¬μ©ν• κ²½μ°
            min_val = min(min_val, abs(curr_s - curr_b))
        return

    # ν„μ¬ μ¬λ£λ¥Ό μ‚¬μ©ν•λ” κ²½μ°
    dfs(idx + 1, curr_s * S[idx], curr_b + B[idx], True)

    # ν„μ¬ μ¬λ£λ¥Ό μ‚¬μ©ν•μ§€ μ•λ” κ²½μ°
    dfs(idx + 1, curr_s, curr_b, used)

dfs(0, 1, 0, False)
