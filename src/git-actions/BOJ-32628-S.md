```python
import sys
input = sys.stdin.readline

N, K = map(int, input().split())
A = [0] + list(map(int, input().split()))
B = [0] + list(map(int, input().split()))

# λ„μ ν•©
for i in range(1, N + 1):
    A[i] += A[i - 1]
    B[i] += B[i - 1]

ans = float('inf')
for i in range(N + 1):
    j = 2 * N - K - i
    if 0 <= j <= N:
        ans = min(ans, max(A[i], B[j]))

print(ans)

```
