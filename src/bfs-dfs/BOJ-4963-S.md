```python
while 1: 
    w, h = map(int, input().split())
    if w == 0 and h == 0:
        exit()
    
    maps = []
    for _ in range(h):
        maps.append(list(map(int, input().split())))
    
    dx = [0, 0, -1, 1, -1, -1, 1, 1]
    dy = [-1, 1, 0, 0, -1, 1, -1, 1]
    count = 0

    def dfs(nx, ny):
        if not (0 <= nx < h and 0 <= ny < w) or maps[nx][ny] == 0:
            return

        maps[nx][ny] = 0
        for i in range(8):
            dfs(nx + dx[i], ny + dy[i])

    
    for i in range(h):
        for j in range(w):
            if maps[i][j] == 1:
                dfs(i, j)
                count+=1
    print(count)

```
