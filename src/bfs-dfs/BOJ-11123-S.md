import sys
from collections import deque

sys.setrecursionlimit(10**6)

N = int(input())

for _ in range(N):
    h, w = map(int, input().split())

    dx = [0, 0, -1, 1]
    dy = [-1, 1, 0, 0]

    maps = []
    for _ in range(h):
        maps.append(list(input()))

    def bfs(x, y):
        queue = deque()
        queue.append((x, y))
        maps[y][x] = '.'  # ë°©ë¬¸ ì²˜ë¦¬

        while queue:
            cx, cy = queue.popleft()
            for i in range(4):
                nx = cx + dx[i]
                ny = cy + dy[i]
                if 0 <= nx < w and 0 <= ny < h and maps[ny][nx] == '#':
                    maps[ny][nx] = '.'
                    queue.append((nx, ny))

    count = 0
    for i in range(h):
        for j in range(w):
            if maps[i][j] == '#':
                bfs(j, i)
                count += 1

    print(count)
