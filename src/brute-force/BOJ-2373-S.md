from collections import defaultdict

N = int(input())
files = []
max_len = 0
for _ in range(N):
    file = list(map(int, input().split()))
    files.append(file[:-1])
    max_len = max(max_len, len(files[-1]))

for k in range(1, max_len+1):
    seen = set()
    for file in files:
        prefix = tuple(file[:k] + [0] * (k-len(file)))
        seen.add(prefix)
    if len(seen) == N:
        print(k)
