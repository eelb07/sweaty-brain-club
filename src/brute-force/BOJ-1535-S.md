N = int(input())

hp = list(map(int, input().split()))
gr = list(map(int, input().split()))

max_gr = 0 
def dfs(idx, curr_h, curr_g):
    global max_gr
    
    if idx == N:
        max_gr = max(max_gr, curr_g)
        return
    
    if curr_h - hp[idx] > 0:
        dfs(idx + 1, curr_h - hp[idx], curr_g + gr[idx])
    dfs(idx + 1, curr_h , curr_g )
dfs(0, 100, 0)
print(max_gr)
