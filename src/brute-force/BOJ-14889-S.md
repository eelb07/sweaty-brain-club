from itertools import combinations

N = int(input())
maps = []
for _ in range(N):
    maps.append(list(map(int, input().split())))


mem = []
for i in range(N):
    mem.append(i)

mem_set = set(mem)
min_abs = float('inf')

for comb in combinations(mem, N//2):  # íŒ€ ì¡°í•©
    team_A = list(comb)
    team_B = list(mem_set - set(comb))
    sum_A = 0
    sum_B = 0 
    # AíŒ€ ëŠ¥ë ¥ì¹˜ ê³„ì‚°
    for i, j in combinations(team_A, 2):
        sum_A += maps[i][j] + maps[j][i]

    # BíŒ€ ëŠ¥ë ¥ì¹˜ ê³„ì‚°
    for i, j in combinations(team_B, 2):
        sum_B += maps[i][j] + maps[j][i]


    diff = abs(sum_A - sum_B)

    if diff < min_abs:
        min_abs = diff

print(min_abs)
